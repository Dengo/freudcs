Methods

$(".test1").animatedBanner({ jsonObject }); --> Initialize banner
$(".test1").animatedBanner("reset"); --> Resets banner, starts animations again
$(".test1").animatedBanner("stop"); --> Stops animation
$(".test1").animatedBanner("destroy"); --> Destroys animations and all data related
